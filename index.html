// 🔗 เปลี่ยนตรงนี้ให้เป็น URL ใหม่ที่คุณเพิ่งก๊อปมา const URL_API = "ใส่_URL_เว็บแอป_ตัวใหม่ล่าสุดที่นี่"; async function loadData() { try { const res = await fetch(URL_API); const data = await res.json(); // ตรวจสอบว่ามีข้อมูลจริงไหม (data[0] คือหัวตาราง) if (data && data.length > 1) { renderDashboard(data); } } catch (e) { console.error("ดึงข้อมูลไม่สำเร็จ:", e); } } function renderDashboard(allRows) { const today = new Date().toISOString().split('T')[0]; const thisMonth = today.substring(0, 7); let tSum = 0; let mSum = 0; // เริ่มวนลูปที่แถวที่ 1 (ข้ามหัวข้อ) for (let i = 1; i < allRows.length; i++) { const row = allRows[i]; const dateStr = new Date(row[0]).toISOString().split('T')[0]; const price = parseFloat(row[2]) || 0; // ยอดเงินอยู่คอลัมน์ที่ 3 const monthStr = row[3]; // เดือนอยู่คอลัมน์ที่ 4 if (dateStr === today) tSum += price; if (monthStr === thisMonth) mSum += price; } // แสดงผลบนหน้าจอ document.getElementById('tSum').innerText = tSum.toLocaleString(); document.getElementById('mSum').innerText = mSum.toLocaleString(); }
